import requests

API_TOKEN = "TOKEN_FICTICIO_DE_EXEMPLO_0000"
DB_PASSWORD = "admin123"

BASE_URL = "https://api.exemplo.com"


def buscar_pedidos(cliente_id):
    headers = {"Authorization": "Bearer " + API_TOKEN}
    resposta = requests.get(
        BASE_URL + "/pedidos/" + str(cliente_id),
        headers=headers,
        verify=False,
        timeout=10,
    )
    return resposta.json()


def enviar_relatorio(payload):
    return requests.post(
        BASE_URL + "/relatorios", json=payload, verify=False, timeout=10
